                            # chunk

                            const chunk = (arr, size) => {
  if (!size) {
    return arr
  }
  const length = Math.ceil(arr.length / size)
  const newArr = Array.from({length}, () => [])
  arr.forEach((value, index) => {
    const newArrIndex = Math.floor(index / size)
    newArr[newArrIndex].push(value)
  })
  return newArr
}

module.exports = chunk

                            # memoize

                            const memoize = (fn) => {
  const cache = {}
  return (n) => {
    if (!cache[n]) {
      cache[n] = fn(n)
    }
    return cache[n]
  }
}

                            # once

                            type OnceCallback<T> = (value?: T) => void
export const once = function<T>(callback: OnceCallback<T>): OnceCallback<T> {
  let isCalled = false
  return function(value?: T): void {
    if (!isCalled) {
      isCalled = true
      callback(value)
    }
  }
}

const onceFn = once(console.log)
onceFn('test')
onceFn('test')
onceFn('test')
onceFn('test')

                            # pipe

                            const pipe = (fn, ...fns) => (...arg) => {
  return fns.reduce((acc, fn) => fn(acc), fn(...arg))
}

  v => v * 10,

  console.log
)(10)

const clone = (obj) => {
  if (typeof (obj) !== 'object') {
    return obj
  }

  var copiedObj = obj.constructor()

  for (var attr in obj) {
    if (obj.hasOwnProperty(attr)) {
      copiedObj[attr] = clone(obj[attr])
    }
  }

  return copiedObj
}
const merge = (defaultOptions, _options) => {
  var newOptions = clone(_options)

  for (var keyName in defaultOptions) {
    if (isUndefined(newOptions[keyName])) {
      newOptions[keyName] = defaultOptions[keyName]
    }
  }

  return newOptions
}
